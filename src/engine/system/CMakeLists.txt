aux_source_directory(${CMAKE_CURRENT_SOURCE_DIR} PROJECT_SOURCES_SYSTEM)

target_include_directories(${PROJECT_NAME}
	PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}
)

if(WIN32)
	aux_source_directory(${CMAKE_CURRENT_SOURCE_DIR}/win PROJECT_SOURCES_SYSTEM_PLATFORM)
elseif(UNIX)
	aux_source_directory(${CMAKE_CURRENT_SOURCE_DIR}/unix PROJECT_SOURCES_SYSTEM_PLATFORM)
else()
	aux_source_directory(${CMAKE_CURRENT_SOURCE_DIR}/null PROJECT_SOURCES_SYSTEM_PLATFORM)
endif()

target_sources(${PROJECT_NAME}
	PRIVATE ${PROJECT_SOURCES_SYSTEM}
	PRIVATE ${PROJECT_SOURCES_SYSTEM_PLATFORM}
)
